CREATE SEQUENCE DOMINIO.SEQ_PAIS INCREMENT BY 1 START 1;

CREATE TABLE DOMINIO.TB_PAIS(

	ID_PAIS NUMERIC NOT NULL,
	CODIGO_PAIS VARCHAR(3) NOT NULL,
	NOME_PAIS VARCHAR(255) NOT NULL,
	CONSTRAINT ID_PAIS_PK PRIMARY KEY(ID_PAIS)

);

CREATE SEQUENCE DOMINIO.SEQ_UNIDADE_FEDERATIVA INCREMENT BY 1 START 1;

CREATE TABLE DOMINIO.TB_UF(

	ID_UF NUMERIC NOT NULL,
	CODIGO_UF VARCHAR(3) NOT NULL,
	NOME_UF VARCHAR(255) NOT NULL,
	ID_PAIS NUMERIC NOT NULL,
	CONSTRAINT ID_UF_PK PRIMARY KEY(ID_UF),
	CONSTRAINT TB_UF_ID_PAIS_FK FOREIGN KEY(ID_PAIS) REFERENCES DOMINIO.TB_PAIS(ID_PAIS)
);

CREATE SEQUENCE DOMINIO.SEQ_CIDADE INCREMENT BY 1 START 1;

CREATE TABLE DOMINIO.TB_CIDADE(

	ID_CIDADE NUMERIC NOT NULL,
	NOME_CIDADE VARCHAR(255) NOT NULL,
	ID_UF NUMERIC NOT NULL,
	CONSTRAINT ID_CIDADE_PK PRIMARY KEY(ID_CIDADE),
	CONSTRAINT TB_CIDADE_ID_UF_FK FOREIGN KEY(ID_UF) REFERENCES DOMINIO.TB_UF(ID_UF)
);

ALTER TABLE DOMINIO.TB_UF
ALTER COLUMN CODIGO_UF TYPE CHAR(2)

ALTER TABLE DOMINIO.TB_PAIS
ALTER COLUMN CODIGO_PAIS TYPE CHAR(3)
